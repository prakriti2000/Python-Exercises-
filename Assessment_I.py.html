40   Q2. Dictionary Aggregation 
41   Given a list of dictionaries representing sales transactions: 
42   sales = [ 
43     {"product": "Pen", "amount": 10}, 
44     {"product": "Book", "amount": 20}, 
45     {"product": "Pen", "amount": 15}, 
46     {"product": "Pencil", "amount": 5} 
47   ] 
48   Write a Python program to calculate total sales per product and print the result as: 
49   Pen: 25 
50   Book: 20 
51   Pencil: 5 
52   """
53   sales = [
54     {"product": "Pen", "amount": 10},
55     {"product": "Book", "amount": 20},
56     {"product": "Pen", "amount": 15},
57     {"product": "Pencil", "amount": 5}
58   ]
59   
60   total = {}
61   
62   for item in sales:
63       p = item["product"]
64       total[p] = total.get(p, 0) + item["amount"]
65   
66   for product, total in total.items():
67       print(f"{product}: {total}")

69   """ 
70   Q3. Unique Numbers in List 
71   Write a Python function that takes a list of integers and returns a new list containing elements that appear exactly once. 
72   Example: 
73   Input: [4, 5, 4, 6, 7, 5, 8] 
74   Output: [6, 7, 8] 
75   """
76   def unique_numbers(nums):
77       counts = {}
78       for n in nums:
79           counts[n] = counts.get(n, 0) + 1
80       return [n for n in nums if counts[n] == 1]
